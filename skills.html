// Node.js Redis Cache for “Daily Mix”
app.get("/daily-mix/:userId", async (req, res) => {
  const cachedMix = await redisClient.get(`mix:${req.params.userId}`);
  if (cachedMix) return res.json(JSON.parse(cachedMix));

  const songs = await db.query("SELECT * FROM recommendations WHERE user_id=?", [req.params.userId]);
  await redisClient.setEx(`mix:${req.params.userId}`, 86400, JSON.stringify(songs)); // cache for 24 hrs
  res.json(songs);
});
